Anonymous Functions

Home Features

  • Do not have an identifier ( a name ),
  • Are usually used as callback functions;
    meaning, they can be passed as function parameters

IIFE

var myObj = (function() {
  var myVar = "a value here";
  function myFn() {
    console.log(myVar);
  }
  var myObj = {};
  myObj.show = function() {
    myFn();
  }
return myObj;
})();
myObj.show();
